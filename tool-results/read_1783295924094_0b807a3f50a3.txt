   135→function formatDate(d?: string | null): string {
   136→  if (!d) return "-";
   137→  try {
   138→    return new Date(d).toLocaleDateString("id-ID", { day: "2-digit", month: "long", year: "numeric" });
   139→  } catch {
   140→    return d;
   141→  }
   142→}
   143→
   144→function formatDateTime(d?: string | null): string {
   145→  if (!d) return "-";
   146→  try {
   147→    return new Date(d).toLocaleString("id-ID", {
   148→      day: "2-digit", month: "short", year: "numeric",
   149→      hour: "2-digit", minute: "2-digit",
   150→    });
   151→  } catch {
   152→    return d;
   153→  }
   154→}
   155→
   156→function formatBytes(n?: number | null): string {
   157→  if (!n) return "-";
   158→  if (n < 1024) return `${n} B`;
   159→  if (n < 1024 * 1024) return `${(n / 1024).toFixed(1)} KB`;
   160→  return `${(n / (1024 * 1024)).toFixed(2)} MB`;
   161→}

   163→function getFileIcon(name: string) {
   164→  const ext = name.split(".").pop()?.toLowerCase() || "";
   165→  if (ext === "pdf") return FileType2;
   166→  if (["jpg", "jpeg", "png", "gif", "webp", "bmp", "svg"].includes(ext)) return FileImage;
   167→  return Files;
   168→}

   203→function RiwayatTanahCard({
   204→  permohonanId,
   205→  items,
   206→  canEdit,
   207→  onRefresh,
   208→}: {
   209→  permohonanId: string;
   210→  items: RiwayatTanahItem[];
   211→  canEdit: boolean;
   212→  onRefresh: () => Promise<void>;
   213→}) {
   214→  const [rtOpen, setRtOpen] = useState(true);
   215→  const [dialogOpen, setDialogOpen] = useState(false);
   216→  const [editingId, setEditingId] = useState<string | null>(null);
   217→  const [form, setForm] = useState<RtDialogForm>(EMPTY_RT_FORM);
   218→  const [saving, setSaving] = useState(false);
   219→  const [deleteId, setDeleteId] = useState<string | null>(null);
   220→  const [deleting, setDeleting] = useState(false);
   221→
   222→  const openAdd = () => {
   223→    setEditingId(null);
   224→    setForm(EMPTY_RT_FORM);
   225→    setDialogOpen(true);
   226→  };
   227→
   228→  const openEdit = (it: RiwayatTanahItem) => {
   229→    setEditingId(it.id);
   230→    setForm({
   231→      tahun: it.tahun || "",
   232→      pemilikSebelumnya: it.pemilikSebelumnya || "",
   233→      hubunganPemilik: it.hubunganPemilik || "Diri Sendiri",
   234→      caraPerolehan: it.caraPerolehan || "",
   235→      noDokumen: it.noDokumen || "",
   236→      keterangan: it.keterangan || "",
   237→    });
   238→    setDialogOpen(true);
   239→  };
   240→
   241→  const handleSave = async () => {
   242→    if (
   243→      !form.tahun?.trim() &&
   244→      !form.pemilikSebelumnya?.trim() &&
   245→      !form.caraPerolehan &&
   246→      !form.keterangan?.trim()
   247→    ) {
   248→      toast.error("Isi minimal salah satu field riwayat tanah");
   249→      return;
   250→    }
   251→    setSaving(true);
   252→    try {
   253→      const body: any = {
   254→        tahun: form.tahun?.trim() || undefined,
   255→        pemilikSebelumnya: form.pemilikSebelumnya?.trim() || undefined,
   256→        hubunganPemilik: form.hubunganPemilik || undefined,
   257→        caraPerolehan: form.caraPerolehan || undefined,
   258→        noDokumen: form.noDokumen?.trim() || undefined,
   259→        keterangan: form.keterangan?.trim() || undefined,
   260→      };
   261→      if (editingId) {
   262→        await api.updateRiwayatTanah(permohonanId, editingId, body);
   263→        toast.success("Riwayat tanah diperbarui");
   264→      } else {
   265→        await api.addRiwayatTanah(permohonanId, body);
   266→        toast.success("Riwayat tanah ditambahkan");
   267→      }
   268→      setDialogOpen(false);
   269→      await onRefresh();
   270→    } catch (e: any) {
   271→      toast.error(e?.message || "Gagal menyimpan riwayat tanah");
   272→    } finally {
   273→      setSaving(false);
   274→    }
   275→  };
   276→
   277→  const handleDelete = async () => {
   278→    if (!deleteId) return;
   279→    setDeleting(true);
   280→    try {
   281→      await api.deleteRiwayatTanah(permohonanId, deleteId);
   282→      toast.success("Riwayat tanah dihapus");
   283→      setDeleteId(null);
   284→      await onRefresh();
   285→    } catch (e: any) {
   286→      toast.error(e?.message || "Gagal menghapus riwayat tanah");
   287→    } finally {
   288→      setDeleting(false);
   289→    }
   290→  };
   291→
   292→  return (
   293→    <Collapsible open={rtOpen} onOpenChange={setRtOpen}>
   294→      <Card className="glass-card border-primary/15">
   295→        <CardContent className="p-5">
   296→          <CollapsibleTrigger asChild>
   297→            <h3 className="font-semibold text-sm flex items-center gap-2 mb-3 cursor-pointer select-none">
   298→              <History className="w-4 h-4 text-primary" /> Riwayat Tanah
   299→              <Badge variant="secondary" className="ml-1 text-[10px]">{items.length}</Badge>
   300→              <ChevronDown className={`w-4 h-4 text-muted-foreground transition-transform duration-200 ${rtOpen ? "rotate-180" : "rotate-0"}`} />
   301→              {canEdit && (
   302→                <Button
   303→                  variant="outline"
   304→                  size="sm"
   305→                  className="ml-auto h-7 px-2 text-[11px] border-primary/30 text-primary hover:bg-primary/5"
   306→                  onClick={(e) => {
   307→                    e.stopPropagation();
   308→                    openAdd();
   309→                  }}
   310→                >
   311→                  <Plus className="w-3.5 h-3.5" /> Tambah
   312→                </Button>
   313→              )}
   314→            </h3>
   315→          </CollapsibleTrigger>
   316→          <Separator className="mb-3" />
   317→          <CollapsibleContent>
   318→            {items.length === 0 ? (
   319→              <div className="text-center py-6 px-4 rounded-lg bg-muted/30 border border-dashed border-border/60">
   320→                <Landmark className="w-7 h-7 text-muted-foreground/60 mx-auto mb-2" />
   321→                <p className="text-xs text-muted-foreground mb-1">Belum ada riwayat tanah</p>
   322→                <p className="text-[11px] text-muted-foreground/70">
   323→                  {canEdit
   324→                    ? "Klik \"Tambah\" untuk mencatat riwayat perolehan dan kepemilikan tanah."
   325→                    : "Petugas/admin belum mencatat riwayat tanah untuk permohonan ini."}
   326→                </p>
   327→              </div>
   328→            ) : (
   329→              <div className="space-y-2.5">
   330→                {/* Timeline-style list */}
   331→                {items.map((it, idx) => (
   332→                  <div
   333→                    key={it.id}
   334→                    className="relative pl-8 pr-3 py-3 rounded-lg border border-border/60 bg-background/40 hover:border-primary/30 transition-colors"
   335→                  >
   336→                    {/* Timeline dot + connector */}
   337→                    <div className="absolute left-3 top-3.5 w-3 h-3 rounded-full bg-primary border-2 border-background" />
   338→                    {idx < items.length - 1 && (
   339→                      <div className="absolute left-[15px] top-7 bottom-0 w-px bg-border/60" />
   340→                    )}
   341→                    <div className="flex items-start justify-between gap-2">
   342→                      <div className="flex-1 min-w-0">
   343→                        <div className="flex flex-wrap items-center gap-1.5 mb-1.5">
   344→                          <Badge variant="outline" className="text-[10px] border-primary/30 bg-primary/5 text-primary">
   345→                            #{it.urutan || idx + 1}
   346→                          </Badge>
   347→                          {it.tahun && (

   349→                              <Calendar className="w-3 h-3 mr-1" /> {it.tahun}
